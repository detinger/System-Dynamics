def DOI(state, t):
    x = state[0]
    y = state[1]

    TP = 10000
    AF = 0.015
    AE = 0.011
    CR = 100

    AFA = (x * AE)
    AFWOM = (x * CR * AF * (y/TP))
    AR = AFA + AFWOM

    dAD = AR
    dPA = -AR
    return [dPA, dAD]

t = np.arange(0, 8, 0.1)
state0 = [10000, 0]
state = odeint(DOI, state0, t)
